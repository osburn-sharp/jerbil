def initialize
  
  @my_service = Jerbil::Service.new(@name, @env, :verify_callback, :stop_callback)
  @key = @my_service.key
  # register the service
  # and start it
  DRb.start_service("druby://#{@my_service.address}", self)
  puts "Started service on #{@my_service.address}"

def stop_callback(key="")

  # remove the service
  exit!
