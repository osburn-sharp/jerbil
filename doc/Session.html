def find(args)
  options = {:name=>nil, :port=>nil, :env=>nil, :pid=>nil}.merge(args)
  pattern = Jerbil::ServiceRecord.new(options[:name], options[:port], options[:env], options[:pid])
  results = Array.new
  @store.each do |service|
    results << service if service == pattern
  end
  return results
end
